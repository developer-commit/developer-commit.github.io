#include <iostream>
#include <vector>
using namespace std;

vector<vector<int>> snail(int N) {
    vector<vector<int>> board(N, vector<int>(N, 0));
    int dx[4] = {0, 1, 0, -1}; // ìš°, í•˜, ì¢Œ, ìƒ
    int dy[4] = {1, 0, -1, 0};
    int x = 0, y = 0, dir = 0;

    for (int num = 1; num <= N * N; ++num) {
        board[x][y] = num;
        int nx = x + dx[dir];
        int ny = y + dy[dir];

        if (nx >= 0 && nx < N && ny >= 0 && ny < N && board[nx][ny] == 0) {
            x = nx;
            y = ny;
        } else {
            dir = (dir + 1) % 4;
            x += dx[dir];
            y += dy[dir];
        }
    }

    return board;
}

int main() {
    int N = 3;
    auto result = snail(N);
    for (const auto& row : result) {
        for (int num : row) {
            cout << num << " ";
        }
        cout << endl;
    }
    return 0;
}

#include <iostream>

vector<vector<int>> solution(int n) {
    vector<vector<int>> answer(n, vector<int>(n));

    int x = 0;
    int loc_x = -1, loc_y = 0;
    bool flag = true;

    for(int i = n ; i > 0; ) {
        for(int j = 0; j < i ; j++) {
            if (flag) answer[loc_y][++loc_x] = ++x;
            else answer[loc_y][--loc_x] = ++x;
        }
        i--;
        if (x == n*n) break;
        flag = !flag;

        for(int j = 0; j < i ; j++) {
            if (flag)  answer[--loc_y][loc_x] = ++x;
            else answer[++loc_y][loc_x] = ++x;
        }
    }

    return answer;
}
